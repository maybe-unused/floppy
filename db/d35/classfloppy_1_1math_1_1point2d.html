template<typename U = default_unit, concepts::num T = f32>
class floppy::math::point2d< U, T >

A two-dimensional point tagged with a unit.

Template Parameters
UAssociated unit of measurement. Default is default_unit.
TNumber type. Must satisfy concept floppy::concepts::num. Default is f32.

constexpr auto lerp ( point2d< U, T > const & other,
T t ) const -> point2d<U, T>
inlineconstexpr

Linearly interpolate between this point and another point.

Example:

using floppy::math::point2d;
auto const from = point2d(0.0F, 10.0F);
auto const to = point2d(8.0F, -4.0F);
fmt::println("{}", from.lerp(to, -1.0F));
fmt::println("{}", from.lerp(to, 0.0F));
fmt::println("{}", from.lerp(to, 0.5F));
fmt::println("{}", from.lerp(to, 1.0F));
fmt::println("{}", from.lerp(to, 2.0F));
