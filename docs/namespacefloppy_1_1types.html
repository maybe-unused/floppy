Typedef documentation

template<std::destructible T>
using floppy::types::option = std::optional<T>

Safe alias for std::optional.

Defaults to std::optional if available, otherwise std::experimental::optional. Example usage with helper functions:

template <typename T>
auto first(std::vector<T> const& vec) -> option<T> {
  if(vec.empty())
   return none;
  return some(vec.front());
}
