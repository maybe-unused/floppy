template<typename T>
using floppy::traits::pimpl = std::experimental::propagate_const<std::unique_ptr<T>>

Pointer-to-implementation pattern trait.

Template parameters
T Underlying type.

Allows to use Pimpl pattern in custom types and classes. Example declaration:

class TestPimpl : public traits::pin<TestPimpl>
{
 public:
  TestPimpl() = default;

  [[nodiscard]] auto add(int a, int b) const -> int;

 private:
  struct Impl;
  traits::pimpl<struct Impl> impl;
};

 struct TestPimpl::Impl {
  auto add(int a, int b) const -> int { return a + b; }
};

 auto TestPimpl::add(int a, int b) const -> int {
  return this->impl->add(a, b);
}
