15def cmake_version(directory: str) -> str:
16 print(colored(f'▶ searching for CMakeLists.txt in {directory}', "magenta"))
17 try:
18 with open(os.path.join(directory, "CMakeLists.txt"), "r") as f:
19 lines = f.readlines()
20 # use regex: VERSION (\d+\.)?(\d+\.)?(\*|\d+)
21 for i, line in enumerate(lines):
22 if re.search(r"VERSION (\d+\.)?(\d+\.)?(\*|\d+)", line) and not line.startswith("cmake_"):
23 version = re.search(r"VERSION (\d+\.)?(\d+\.)?(\*|\d+)", line).group(0)
24 version = version.replace("VERSION ", "")
25 print(colored(f"▶ found CMakeLists.txt version: {version}", "green"))
26 return version
27 print(colored("▶ could not find CMakeLists.txt version", "red"))
28 return None
29 except FileNotFoundError:
30 raise FileNotFoundError(f"CMakeLists.txt not found in {directory}")
