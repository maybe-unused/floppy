163 [[nodiscard]] static constexpr auto swap_endian(T u) noexcept -> T {
164 static_assert(std::numeric_limits<unsigned char>::digits == 8, "unsigned char is not 8 bits"); // NOLINT(*-magic-numbers)
165 union {
166 T u;
167 u8 c[sizeof(T)]; // NOLINT(*-avoid-c-arrays)
168 } source {}, dest {};
169 source.u = u;
170 for(auto k = 0_ZU; k < sizeof(T); k++)
171 dest.c[k] = source.c[sizeof(T) - k - 1];
172 return dest.u;
173 }
