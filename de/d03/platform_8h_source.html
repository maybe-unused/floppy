248 template <typename T>
249 auto swap_endian(T u) -> T {
250 static_assert(std::numeric_limits<unsigned char>::digits == 8, "unsigned char is not 8 bits");
251 union {
252 T u;
253 u8 c[sizeof(T)];
254 } source {}, dest {};
255 source.u = u;
256 for(auto k = 0_ZU; k < sizeof(T); k++)
257 dest.c[k] = source.c[sizeof(T) - k - 1];
258 return dest.u;
259 }
