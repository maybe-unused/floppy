112 auto configuration_file<F, T>::write_to_file(std::string_view content) const noexcept(false) -> void
113 {
114 namespace fs = std::filesystem;
115
116 if(not fs::exists(this->path().parent_path()))
117 fs::create_directories(this->path().parent_path());
118 auto handle = std::ofstream(this->path());
119 if(not handle.is_open())
120 throw std::runtime_error("failed to open file handle for writing at " + this->path().string());
121 if(fs::exists(this->path())) {
122 auto const permissions = fs::perms::owner_read
123 | fs::perms::owner_write
124 | fs::perms::group_read
125 | fs::perms::others_read
126 | fs::perms::others_exec;
127 fs::permissions(this->path(), permissions);
128 log::trace("config: rw file permissions set to: {}", this->path().string());
129 }
130 handle << content;
131 if(not handle.good())
132 throw std::runtime_error("failed to write to file at " + this->path().string());
133 log::trace("config: file ({} bytes) written to path: {}", content.size(), this->path().string());
