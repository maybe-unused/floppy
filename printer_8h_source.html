128 auto print_trace(std::FILE* out, resolved_trace const& trace) -> void {
129 printer::indent(out, trace.idx);
130 auto already_indented = true;
131 if(trace.source.file_name().empty() or this->object) {
132 fmt::print(out, R"( Object '{}' at '{}' in '{}')",
133 fmt::styled(trace.object_filename, fmt::emphasis::bold | fg(fmt::terminal_color::white)),
134 fmt::styled(trace.addr, fmt::emphasis::faint | fg(fmt::terminal_color::white)),
135 fmt::styled(trace.object_function, fg(fmt::terminal_color::yellow))
136 );
137 fmt::print(out, "\n");
138 already_indented = false;
139 }
140
141 for(size_t inliner_idx = trace.inliners.size(); inliner_idx > 0; --inliner_idx) {
142 if(not already_indented)
143 fmt::print(out, " ");
144 auto const& inliner_loc = trace.inliners[inliner_idx - 1];
145 this->print_source_loc(out, " | ", inliner_loc);
146 if(snippet)
147 this->print_snippet(out, " | ", inliner_loc, inliner_context_size);
148 already_indented = false;
149 }
150
151 if(not trace.source.file_name().empty()) {
152 if(not already_indented)
153 fmt::print(out, " ");
154 this->print_source_loc(out, " ", trace.source, trace.addr);
155 if(snippet)
156 this->print_snippet(out, " ", trace.source, trace_context_size);
157 }
