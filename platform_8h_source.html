188 template <typename C>
189 [[nodiscard]] auto as_target_triple() const noexcept -> std::basic_string<C> {
190 return fmt::format(
191 "{}-{}-{}{}",
192 (std::basic_stringstream<C>() << this->architecture).str(),
193 (std::basic_stringstream<C>() << platform::guess_vendor(this->operating_system)).str(),
194 (std::basic_stringstream<C>() << this->operating_system).str(),
195 this->compiler == compiler::unknown
196 or (this->operating_system != operating_system::windows
197 and this->operating_system != operating_system::gnu_linux)
198 ? ""
199 : fmt::format("-{}", (std::basic_stringstream<C>() << this->compiler).str())
200 );
201 }

207 template <typename T>
208 [[nodiscard]] static constexpr auto swap_endian(T u) noexcept -> T {
209 static_assert(std::numeric_limits<unsigned char>::digits == 8, "unsigned char is not 8 bits"); // NOLINT(*-magic-numbers)
210 union {
211 T u;
212 u8 c[sizeof(T)]; // NOLINT(*-avoid-c-arrays)
213 } source {}, dest {};
214 source.u = u;
215 for(auto k = 0_ZU; k < sizeof(T); k++)
216 dest.c[k] = source.c[sizeof(T) - k - 1];
217 return dest.u;
218 }
