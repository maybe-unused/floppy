59 resolved_trace resolve(resolved_trace trace) override {
60 using namespace details;
61 Dwarf_Addr trace_addr = reinterpret_cast<Dwarf_Addr>(trace.addr);
62 if (!_dwfl_handle_initialized) {
63 _dwfl_cb.reset(new Dwfl_Callbacks);
64 _dwfl_cb->find_elf = &dwfl_linux_proc_find_elf;
65 _dwfl_cb->find_debuginfo = &dwfl_standard_find_debuginfo;
66 _dwfl_cb->debuginfo_path = 0;
67 _dwfl_handle.reset(dwfl_begin(_dwfl_cb.get()));
68 _dwfl_handle_initialized = true;
69
70 if(not _dwfl_handle)
71 return trace;
72
73 dwfl_report_begin(_dwfl_handle.get());
74 int r = dwfl_linux_proc_report(_dwfl_handle.get(), getpid());
75 dwfl_report_end(_dwfl_handle.get(), NULL, NULL);
76 if(r < 0)
77 return trace;
78 }
79
80 if(not _dwfl_handle)
81 return trace;
82 Dwfl_Module *mod = dwfl_addrmodule(_dwfl_handle.get(), trace_addr);
83 if(mod) {
84 const char *module_name = dwfl_module_info(mod, 0, 0, 0, 0, 0, 0, 0);
85 if(module_name)
86 trace.object_filename = module_name;
87
88 const char *sym_name = dwfl_module_addrname(mod, trace_addr);
89 if(sym_name)
90 trace.object_function = demangle(sym_name);
91 }
92
93 Dwarf_Addr mod_bias = 0;
94 Dwarf_Die *cudie = dwfl_module_addrdie(mod, trace_addr, &mod_bias);
95 if(not cudie) {
96 while((cudie = dwfl_module_nextcu(mod, cudie, &mod_bias))) {
97 Dwarf_Die die_mem;
98 Dwarf_Die *fundie = find_fundie_by_pc(cudie, trace_addr - mod_bias, &die_mem);
99 if(fundie)
100 break;
101 }
102 }
103
104 if(not cudie)
105 return trace; // this time we lost the game :/
106
107 // Now that we have a compilation unit DIE, this function will be able
108 // to load the corresponding section in .debug_line (if not already
109 // loaded) and hopefully find the source location mapped to our
110 // address.
111 Dwarf_Line *srcloc = dwarf_getsrc_die(cudie, trace_addr - mod_bias);
112
113 if (srcloc) {
114 const char *srcfile = dwarf_linesrc(srcloc, 0, 0);
115 if (srcfile) {
116 trace.source.file_name_mut() = srcfile;
117 }
118 int line = 0, col = 0;
119 dwarf_lineno(srcloc, &line);
120 dwarf_linecol(srcloc, &col);
121 trace.source.line_mut() = static_cast<unsigned>(line);
122 trace.source.column_mut() = static_cast<unsigned>(col);
123 }
124 deep_first_search_by_pc(cudie, trace_addr - mod_bias, inliners_search_cb(trace));
125 if(trace.source.function_name().empty())
126 trace.source.function_name_mut() = trace.object_function;
127 return trace;
128 }

169 static bool die_has_pc(Dwarf_Die *die, Dwarf_Addr pc) {
170 Dwarf_Addr low, high;
171
172 // continuous range
173 if (dwarf_hasattr(die, DW_AT_low_pc) && dwarf_hasattr(die, DW_AT_high_pc)) {
174 if(dwarf_lowpc(die, &low) != 0)
175 return false;
176 if(dwarf_highpc(die, &high) != 0) {
177 Dwarf_Attribute attr_mem;
178 Dwarf_Attribute *attr = dwarf_attr(die, DW_AT_high_pc, &attr_mem);
179 Dwarf_Word value;
180 if(dwarf_formudata(attr, &value) != 0)
181 return false;
182 high = low + value;
183 }
184 return pc >= low && pc < high;
185 }
186
187 // non-continuous range.
188 Dwarf_Addr base;
189 ptrdiff_t offset = 0;
190 while ((offset = dwarf_ranges(die, offset, &base, &low, &high)) > 0) {
191 if(pc >= low && pc < high)
192 return true;
193 }
194 return false;
195 }

197 static Dwarf_Die *find_fundie_by_pc(Dwarf_Die *parent_die, Dwarf_Addr pc,
198 Dwarf_Die *result) {
199 if(dwarf_child(parent_die, result) != 0)
200 return 0;
201 Dwarf_Die *die = result;
202 do {
203 switch (dwarf_tag(die)) {
204 case DW_TAG_subprogram:
205 case DW_TAG_inlined_subroutine:
206 if(die_has_pc(die, pc))
207 return result;
208 };
209 bool declaration = false;
210 Dwarf_Attribute attr_mem;
211 dwarf_formflag(dwarf_attr(die, DW_AT_declaration, &attr_mem),
212 &declaration);
213 if(not declaration) {
214 // let's be curious and look deeper in the tree,
215 // function are not necessarily at the first level, but
216 // might be nested inside a namespace, structure etc.
217 Dwarf_Die die_mem;
218 Dwarf_Die *indie = find_fundie_by_pc(die, pc, &die_mem);
219 if(indie) {
220 *result = die_mem;
221 return result;
222 }
223 }
224 } while(dwarf_siblingof(die, result) == 0);
225 return 0;
226 }

228 template <typename CB>
229 static bool deep_first_search_by_pc(Dwarf_Die *parent_die, Dwarf_Addr pc,
230 CB cb) {
231 Dwarf_Die die_mem;
232 if(dwarf_child(parent_die, &die_mem) != 0)
233 return false;
234
235 bool branch_has_pc = false;
236 Dwarf_Die *die = &die_mem;
237 do {
238 bool declaration = false;
239 Dwarf_Attribute attr_mem;
240 dwarf_formflag(dwarf_attr(die, DW_AT_declaration, &attr_mem),
241 &declaration);
242 if (!declaration) {
243 // let's be curious and look deeper in the tree, function are
244 // not necessarily at the first level, but might be nested
245 // inside a namespace, structure, a function, an inlined
246 // function etc.
247 branch_has_pc = deep_first_search_by_pc(die, pc, cb);
248 }
249 if(not branch_has_pc)
250 branch_has_pc = die_has_pc(die, pc);
251 if(branch_has_pc)
252 cb(die);
253 } while(dwarf_siblingof(die, &die_mem) == 0);
254 return branch_has_pc;
255 }

257 static const char* die_call_file(Dwarf_Die *die) {
258 Dwarf_Attribute attr_mem;
259 Dwarf_Word file_idx = 0;
260 dwarf_formudata(dwarf_attr(die, DW_AT_call_file, &attr_mem), &file_idx);
261 if(file_idx == 0)
262 return 0;
263 Dwarf_Die die_mem;
264 Dwarf_Die *cudie = dwarf_diecu(die, &die_mem, 0, 0);
265 if(not cudie)
266 return 0;
267 Dwarf_Files *files = 0;
268 size_t nfiles;
269 dwarf_getsrcfiles(cudie, &files, &nfiles);
270 if(not files)
271 return 0;
272 return dwarf_filesrc(files, file_idx, 0, 0);
273 }

296 resolved_trace resolve(resolved_trace trace) override {
297 // parse:
298 // <n> <file> <addr> <mangled-name> + <offset>
299 char *filename = _symbols[trace.idx];
300
301 // skip "<n> "
302 while(*filename && *filename != ' ')
303 filename++;
304 while(*filename == ' ')
305 filename++;
306
307 // find start of <mangled-name> from end (<file> may contain a space)
308 char *p = filename + strlen(filename) - 1;
309 // skip to start of " + <offset>"
310 while(p > filename && *p != ' ')
311 p--;
312 while(p > filename && *p == ' ')
313 p--;
314 while(p > filename && *p != ' ')
315 p--;
316 while(p > filename && *p == ' ')
317 p--;
318 char *funcname_end = p + 1;
319
320 // skip to start of "<manged-name>"
321 while(p > filename && *p != ' ')
322 p--;
323 char *funcname = p + 1;
324
325 // skip to start of " <addr> "
326 while(p > filename && *p == ' ')
327 p--;
328 while(p > filename && *p != ' ')
329 p--;
330 while(p > filename && *p == ' ')
331 p--;
332
333 // skip "<file>", handling the case where it contains a
334 char *filename_end = p + 1;
335 if(p == filename) {
336 // something went wrong, give up
337 filename_end = filename + strlen(filename);
338 funcname = filename_end;
339 }
340 trace.object_filename.assign(
341 filename, filename_end); // ok even if filename_end is the ending \0
342 // (then we assign entire string)
343
344 if(*funcname) { // if it's not end of string
345 *funcname_end = '\0';
346
347 trace.object_function = this->demangle(funcname);
348 trace.object_function += " ";
349 trace.object_function += (funcname_end + 1);
350 trace.source.function = trace.object_function; // we cannot do better.
351 }

353 }
