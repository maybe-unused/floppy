100class StackTraceImplBase {

103 : _thread_id(0), _skip(0), _context(nullptr), _error_addr(nullptr) {}

105 size_t thread_id() const { return _thread_id; }

146class StackTraceImplHolder : public StackTraceImplBase {
147public:
148 size_t size() const {
149 return (_stacktrace.size() >= skip_n_firsts())
150 ? _stacktrace.size() - skip_n_firsts()
151 : 0;
152 }
153 trace operator[](size_t idx) const {
154 if (idx >= size()) {
155 return trace();
156 }
157 return trace(_stacktrace[idx + skip_n_firsts()], idx);
158 }
159 void *const *begin() const {
160 if (size()) {
161 return &_stacktrace[skip_n_firsts()];
162 }
163 return nullptr;
164 }
165
166protected:
167 std::vector<void *> _stacktrace;
168};

171template <>
172class StackTraceImpl<system_tag::current> : public StackTraceImplHolder {
173public:
174 __noinline__
175 size_t load_here(size_t depth = 32, void *context = nullptr,
176 void *error_addr = nullptr) {
177 set_context(context);
178 set_error_addr(error_addr);
179 load_thread_info();
180 if (depth == 0) {
181 return 0;
182 }
183 _stacktrace.resize(depth + 1);
184 size_t trace_cnt = backtrace(&_stacktrace[0], _stacktrace.size());
185 _stacktrace.resize(trace_cnt);
186 skip_n_firsts(1);
187 return size();
188 }
189
190 size_t load_from(void *addr, size_t depth = 32, void *context = nullptr,
191 void *error_addr = nullptr) {
192 load_here(depth + 8, context, error_addr);
193
194 for (size_t i = 0; i < _stacktrace.size(); ++i) {
195 if (_stacktrace[i] == addr) {
196 skip_n_firsts(i);
197 _stacktrace[i] = (void *)((uintptr_t)_stacktrace[i] + 1);
198 break;
199 }
200 }
201
202 _stacktrace.resize(std::min(_stacktrace.size(), skip_n_firsts() + depth));
203 return size();
204 }

209template <>
210class StackTraceImpl<system_tag::current> : public StackTraceImplHolder {
211public:
212 // We have to load the machine type from the image info
213 // So we first initialize the resolver, and it tells us this info
214 void set_machine_type(DWORD machine_type) { machine_type_ = machine_type; }
215 void set_context(CONTEXT *ctx) { ctx_ = ctx; }
216 void set_thread_handle(HANDLE handle) { thd_ = handle; }
217
218 __noinline__
219 size_t load_here(size_t depth = 32, void *context = nullptr,
220 void *error_addr = nullptr) {
221 set_context(static_cast<CONTEXT*>(context));
222 set_error_addr(error_addr);
223 CONTEXT localCtx; // used when no context is provided
224
225 if (depth == 0) {
226 return 0;
227 }
228
229 if (!ctx_) {
230 ctx_ = &localCtx;
231 RtlCaptureContext(ctx_);
232 }
233
234 if (!thd_) {
235 thd_ = GetCurrentThread();
236 }
237
238 HANDLE process = GetCurrentProcess();
239
240 STACKFRAME64 s;
241 memset(&s, 0, sizeof(STACKFRAME64));
242
243 // TODO: 32 bit context capture
244 s.AddrStack.Mode = AddrModeFlat;
245 s.AddrFrame.Mode = AddrModeFlat;
246 s.AddrPC.Mode = AddrModeFlat;
247#ifdef _M_X64
248 s.AddrPC.Offset = ctx_->Rip;
249 s.AddrStack.Offset = ctx_->Rsp;
250 s.AddrFrame.Offset = ctx_->Rbp;
251#else
252 s.AddrPC.Offset = ctx_->Eip;
253 s.AddrStack.Offset = ctx_->Esp;
254 s.AddrFrame.Offset = ctx_->Ebp;
255#endif
256
257 if (!machine_type_) {
258#ifdef _M_X64
259 machine_type_ = IMAGE_FILE_MACHINE_AMD64;
260#else
261 machine_type_ = IMAGE_FILE_MACHINE_I386;
262#endif
263 }
264
265 for(;;) {
266 // NOTE: this only works if PDBs are already loaded!
267 SetLastError(0);
268 if (!StackWalk64(machine_type_, process, thd_, &s, ctx_, NULL,
269 SymFunctionTableAccess64, SymGetModuleBase64, NULL))
270 break;
271
272 if (s.AddrReturn.Offset == 0)
273 break;
274
275 _stacktrace.push_back(reinterpret_cast<void *>(s.AddrPC.Offset));
276
277 if (size() >= depth)
278 break;
279 }
280
281 return size();
282 }
283
284 size_t load_from(void *addr, size_t depth = 32, void *context = nullptr,
285 void *error_addr = nullptr) {
286 load_here(depth + 8, context, error_addr);
287
288 for (size_t i = 0; i < _stacktrace.size(); ++i) {
289 if (_stacktrace[i] == addr) {
290 skip_n_firsts(i);
291 break;
292 }
293 }
294
295 _stacktrace.resize(std::min(_stacktrace.size(), skip_n_firsts() + depth));
296 return size();
297 }
298
299private:
300 DWORD machine_type_ = 0;
301 HANDLE thd_ = 0;
302 CONTEXT *ctx_ = nullptr;
303};

346class TraceResolverLinuxBase : public TraceResolverImplBase {
347public:
348 TraceResolverLinuxBase()
349 : argv0_(get_argv0()), exec_path_(read_symlink("/proc/self/exe")) {}
350 std::string resolve_exec_path(Dl_info &symbol_info) const {
351 // mutates symbol_info.dli_fname to be filename to open and returns filename
352 // to display
353 if (symbol_info.dli_fname == argv0_) {
354 // dladdr returns argv[0] in dli_fname for symbols contained in
355 // the main executable, which is not a valid path if the
356 // executable was found by a search of the PATH environment
357 // variable; In that case, we actually open /proc/self/exe, which
358 // is always the actual executable (even if it was deleted/replaced!)
359 // but display the path that /proc/self/exe links to.
360 // However, this right away reduces probability of successful symbol
361 // resolution, because libbfd may try to find *.debug files in the
362 // same dir, in case symbols are stripped. As a result, it may try
363 // to find a file /proc/self/<exe_name>.debug, which obviously does
364 // not exist. /proc/self/exe is a last resort. First load attempt
365 // should go for the original executable file path.
366 symbol_info.dli_fname = "/proc/self/exe";
367 return exec_path_;
368 } else
369 return symbol_info.dli_fname;
370 }
371
372private:
373 std::string argv0_;
374 std::string exec_path_;
375
376 static std::string get_argv0() {
377 std::string argv0;
378 std::ifstream ifs("/proc/self/cmdline");
379 std::getline(ifs, argv0, '\0');
380 return argv0;
381 }
382
383 static std::string read_symlink(std::string const &symlink_path) {
384 std::string path;
385 path.resize(100);
386 while (true) {
387 isize len = ::readlink(symlink_path.c_str(), &*path.begin(), path.size());
388 if(len < 0)
389 return "";
390 if (static_cast<size_t>(len) == path.size()) {
391 path.resize(path.size() * 2);
392 } else {
393 path.resize(static_cast<std::string::size_type>(len));
394 break;
395 }
396 }
397
398 return path;
399 }

404template <>
405class TraceResolverLinuxImpl<trace_resolver_tag::libdw>
406 : public TraceResolverLinuxBase {
407public:
408 TraceResolverLinuxImpl() : _dwfl_handle_initialized(false) {}
409
410 resolved_trace resolve(resolved_trace trace) override {
411 using namespace details;
412 Dwarf_Addr trace_addr = reinterpret_cast<Dwarf_Addr>(trace.addr);
413 if (!_dwfl_handle_initialized) {
414 _dwfl_cb.reset(new Dwfl_Callbacks);
415 _dwfl_cb->find_elf = &dwfl_linux_proc_find_elf;
416 _dwfl_cb->find_debuginfo = &dwfl_standard_find_debuginfo;
417 _dwfl_cb->debuginfo_path = 0;
418 _dwfl_handle.reset(dwfl_begin(_dwfl_cb.get()));
419 _dwfl_handle_initialized = true;
420
421 if(not _dwfl_handle)
422 return trace;
423
424 dwfl_report_begin(_dwfl_handle.get());
425 int r = dwfl_linux_proc_report(_dwfl_handle.get(), getpid());
426 dwfl_report_end(_dwfl_handle.get(), NULL, NULL);
427 if(r < 0)
428 return trace;
429 }
430
431 if(not _dwfl_handle)
432 return trace;
433 Dwfl_Module *mod = dwfl_addrmodule(_dwfl_handle.get(), trace_addr);
434 if(mod) {
435 const char *module_name = dwfl_module_info(mod, 0, 0, 0, 0, 0, 0, 0);
436 if(module_name)
437 trace.object_filename = module_name;
438
439 const char *sym_name = dwfl_module_addrname(mod, trace_addr);
440 if(sym_name)
441 trace.object_function = demangle(sym_name);
442 }
443
444 Dwarf_Addr mod_bias = 0;
445 Dwarf_Die *cudie = dwfl_module_addrdie(mod, trace_addr, &mod_bias);
446 if(not cudie) {
447 while((cudie = dwfl_module_nextcu(mod, cudie, &mod_bias))) {
448 Dwarf_Die die_mem;
449 Dwarf_Die *fundie = find_fundie_by_pc(cudie, trace_addr - mod_bias, &die_mem);
450 if(fundie)
451 break;
452 }
453 }
454
455 if(not cudie)
456 return trace; // this time we lost the game :/
457
458 // Now that we have a compilation unit DIE, this function will be able
459 // to load the corresponding section in .debug_line (if not already
460 // loaded) and hopefully find the source location mapped to our
461 // address.
462 Dwarf_Line *srcloc = dwarf_getsrc_die(cudie, trace_addr - mod_bias);
463
464 if (srcloc) {
465 const char *srcfile = dwarf_linesrc(srcloc, 0, 0);
466 if (srcfile) {
467 trace.source.file_name_mut() = srcfile;
468 }
469 int line = 0, col = 0;
470 dwarf_lineno(srcloc, &line);
471 dwarf_linecol(srcloc, &col);
472 trace.source.line_mut() = static_cast<unsigned>(line);
473 trace.source.column_mut() = static_cast<unsigned>(col);
474 }
475 deep_first_search_by_pc(cudie, trace_addr - mod_bias, inliners_search_cb(trace));
476 if(trace.source.function_name().empty())
477 trace.source.function_name_mut() = trace.object_function;
478 return trace;
479 }
480
481private:
482 typedef details::handle<Dwfl *, deleter<void, Dwfl *, &dwfl_end> >
483 dwfl_handle_t;
484 details::handle<Dwfl_Callbacks *, default_delete<Dwfl_Callbacks *> >
485 _dwfl_cb;
486 dwfl_handle_t _dwfl_handle;
487 bool _dwfl_handle_initialized;
488
489 struct inliners_search_cb {
490 void operator()(Dwarf_Die *die) {
491 switch (dwarf_tag(die)) {
492 const char *name;
493 case DW_TAG_subprogram:
494 if((name = dwarf_diename(die)))
495 trace.source.function_name_mut() = name;
496 break;
497
498 case DW_TAG_inlined_subroutine:
499 resolved_trace::source_loc_t sloc;
500 Dwarf_Attribute attr_mem;
501
502 if((name = dwarf_diename(die)))
503 sloc.function_name_mut() = name;
504 if((name = die_call_file(die)))
505 sloc.file_name_mut() = name;
506
507 Dwarf_Word line = 0, col = 0;
508 dwarf_formudata(dwarf_attr(die, DW_AT_call_line, &attr_mem), &line);
509 dwarf_formudata(dwarf_attr(die, DW_AT_call_column, &attr_mem), &col);
510 sloc.line_mut() = static_cast<unsigned>(line);
511 sloc.column_mut() = static_cast<unsigned>(col);
512 trace.inliners.push_back(sloc);
513 break;
514 };
515 }
516 resolved_trace &trace;
517 inliners_search_cb(resolved_trace &t) : trace(t) {}
518 };

520 static bool die_has_pc(Dwarf_Die *die, Dwarf_Addr pc) {
521 Dwarf_Addr low, high;
522
523 // continuous range
524 if (dwarf_hasattr(die, DW_AT_low_pc) && dwarf_hasattr(die, DW_AT_high_pc)) {
525 if(dwarf_lowpc(die, &low) != 0)
526 return false;
527 if(dwarf_highpc(die, &high) != 0) {
528 Dwarf_Attribute attr_mem;
529 Dwarf_Attribute *attr = dwarf_attr(die, DW_AT_high_pc, &attr_mem);
530 Dwarf_Word value;
531 if(dwarf_formudata(attr, &value) != 0)
532 return false;
533 high = low + value;
534 }
535 return pc >= low && pc < high;
536 }
537
538 // non-continuous range.
539 Dwarf_Addr base;
540 ptrdiff_t offset = 0;
541 while ((offset = dwarf_ranges(die, offset, &base, &low, &high)) > 0) {
542 if(pc >= low && pc < high)
543 return true;
544 }
545 return false;
546 }
547
548 static Dwarf_Die *find_fundie_by_pc(Dwarf_Die *parent_die, Dwarf_Addr pc,
549 Dwarf_Die *result) {
550 if(dwarf_child(parent_die, result) != 0)
551 return 0;
552 Dwarf_Die *die = result;
553 do {
554 switch (dwarf_tag(die)) {
555 case DW_TAG_subprogram:
556 case DW_TAG_inlined_subroutine:
557 if(die_has_pc(die, pc))
558 return result;
559 };
560 bool declaration = false;
561 Dwarf_Attribute attr_mem;
562 dwarf_formflag(dwarf_attr(die, DW_AT_declaration, &attr_mem),
563 &declaration);
564 if(not declaration) {
565 // let's be curious and look deeper in the tree,
566 // function are not necessarily at the first level, but
567 // might be nested inside a namespace, structure etc.
568 Dwarf_Die die_mem;
569 Dwarf_Die *indie = find_fundie_by_pc(die, pc, &die_mem);
570 if(indie) {
571 *result = die_mem;
572 return result;
573 }
574 }
575 } while(dwarf_siblingof(die, result) == 0);
576 return 0;
577 }
578
579 template <typename CB>
580 static bool deep_first_search_by_pc(Dwarf_Die *parent_die, Dwarf_Addr pc,
581 CB cb) {
582 Dwarf_Die die_mem;
583 if(dwarf_child(parent_die, &die_mem) != 0)
584 return false;
585
586 bool branch_has_pc = false;
587 Dwarf_Die *die = &die_mem;
588 do {
589 bool declaration = false;
590 Dwarf_Attribute attr_mem;
591 dwarf_formflag(dwarf_attr(die, DW_AT_declaration, &attr_mem),
592 &declaration);
593 if (!declaration) {
594 // let's be curious and look deeper in the tree, function are
595 // not necessarily at the first level, but might be nested
596 // inside a namespace, structure, a function, an inlined
597 // function etc.
598 branch_has_pc = deep_first_search_by_pc(die, pc, cb);
599 }
600 if(not branch_has_pc)
601 branch_has_pc = die_has_pc(die, pc);
602 if(branch_has_pc)
603 cb(die);
604 } while(dwarf_siblingof(die, &die_mem) == 0);
605 return branch_has_pc;
606 }
607
608 static const char* die_call_file(Dwarf_Die *die) {
609 Dwarf_Attribute attr_mem;
610 Dwarf_Word file_idx = 0;
611 dwarf_formudata(dwarf_attr(die, DW_AT_call_file, &attr_mem), &file_idx);
612 if(file_idx == 0)
613 return 0;
614 Dwarf_Die die_mem;
615 Dwarf_Die *cudie = dwarf_diecu(die, &die_mem, 0, 0);
616 if(not cudie)
617 return 0;
618 Dwarf_Files *files = 0;
619 size_t nfiles;
620 dwarf_getsrcfiles(cudie, &files, &nfiles);
621 if(not files)
622 return 0;
623 return dwarf_filesrc(files, file_idx, 0, 0);
624 }

637template <>
638class TraceResolverDarwinImpl<trace_resolver_tag::backtrace_symbol>
639 : public TraceResolverImplBase {
640public:
641 void load_addresses(void *const*addresses, int address_count) override {
642 if(address_count == 0)
643 return;
644 _symbols.reset(backtrace_symbols(addresses, address_count));
645 }
646
647 resolved_trace resolve(resolved_trace trace) override {
648 // parse:
649 // <n> <file> <addr> <mangled-name> + <offset>
650 char *filename = _symbols[trace.idx];
651
652 // skip "<n> "
653 while(*filename && *filename != ' ')
654 filename++;
655 while(*filename == ' ')
656 filename++;
657
658 // find start of <mangled-name> from end (<file> may contain a space)
659 char *p = filename + strlen(filename) - 1;
660 // skip to start of " + <offset>"
661 while(p > filename && *p != ' ')
662 p--;
663 while(p > filename && *p == ' ')
664 p--;
665 while(p > filename && *p != ' ')
666 p--;
667 while(p > filename && *p == ' ')
668 p--;
669 char *funcname_end = p + 1;
670
671 // skip to start of "<manged-name>"
672 while(p > filename && *p != ' ')
673 p--;
674 char *funcname = p + 1;
675
676 // skip to start of " <addr> "
677 while(p > filename && *p == ' ')
678 p--;
679 while(p > filename && *p != ' ')
680 p--;
681 while(p > filename && *p == ' ')
682 p--;
683
684 // skip "<file>", handling the case where it contains a
685 char *filename_end = p + 1;
686 if(p == filename) {
687 // something went wrong, give up
688 filename_end = filename + strlen(filename);
689 funcname = filename_end;
690 }
691 trace.object_filename.assign(
692 filename, filename_end); // ok even if filename_end is the ending \0
693 // (then we assign entire string)
694
695 if(*funcname) { // if it's not end of string
696 *funcname_end = '\0';
697
698 trace.object_function = this->demangle(funcname);
699 trace.object_function += " ";
700 trace.object_function += (funcname_end + 1);
701 trace.source.function = trace.object_function; // we cannot do better.
702 }
703 return trace;
704 }
705
706private:
707 details::handle<char **> _symbols;
708};

860 const std::string &filename_b, unsigned line_b,
861 unsigned context_size) {
862 source_file &src_file_a = get_src_file(filename_a);
863 source_file &src_file_b = get_src_file(filename_b);
864
865 lines_t lines =
866 src_file_a.get_lines(line_a - context_size / 4, context_size / 2);
867 std::ignore = src_file_b.get_lines(line_b - context_size / 4, context_size / 2, lines);

869 }
870
871 lines_t get_coalesced_snippet(const std::string &filename, unsigned line_a,
872 unsigned line_b, unsigned context_size) {
873 source_file &src_file = get_src_file(filename);
874
875 using std::max;
876 using std::min;
877 unsigned a = min(line_a, line_b);
878 unsigned b = max(line_a, line_b);
879
880 if ((b - a) < (context_size / 3)) {
881 return src_file.get_lines((a + b - context_size + 1) / 2, context_size);
882 }
883
884 lines_t lines = src_file.get_lines(a - context_size / 4, context_size / 2);
885 std::ignore = src_file.get_lines(b - context_size / 4, context_size / 2, lines);
886 return lines;
887 }

894 src_files_t::iterator it = _src_files.find(filename);
895 if (it != _src_files.end()) {
896 return it->second;
897 }
