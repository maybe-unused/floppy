188 // \headerfile floppy/logging.h
189 // \ingroup logging
190 // \details
191 // This class is a wrapper around spdlog::logger.
192 // Provides source location information in log messages.
193 // Example usage syntax:
194 // \code {.cpp}
195 // floppy::log::trace()("Message"); // logs message at info level to default logger
196 // floppy::log::trace()("Message {}", 1); // logs message at info level to default logger with 1 as argument
197 // floppy::log::trace()(*spdlog::default_logger(), "Message {} and {}", 1, 2); // logs message at info level to selected logger with 1 and 2 as arguments
198 // \endcode
199 // If you want to log message without source location, pass <code>none</code> to first pair of brackets:
200 // \code {.cpp}
201 // floppy::log::trace(none)("Message {} and {}", 1, 2);
202 // \endcode
203 // \see source_location
204 // \see log
205 // \see debug, info, warn, error, critical
206
208 namespace level {
209 using namespace spdlog::level;
210 } // namespace level
