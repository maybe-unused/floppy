23 [[nodiscard]] constexpr auto hex_to_dec(char const c) -> i32 {
24 return (c >= '0' and c <= '9')
25 ? c - '0'
26 : ((c >= 'a' and c <= 'f')
27 ? c - 'a' + 10
28 : (c >= 'A' and c <= 'F')
29 ? c - 'A' + 10
30 : throw std::invalid_argument("invalid hex character"));
31 }

81 [[nodiscard]] constexpr auto a(f32 const saturation, f32 const lightness) -> f32 {
82 return saturation * min(lightness, 1.0F - lightness);

85 [[nodiscard]] constexpr auto chroma(std::array<u8, 3> const& rgb) -> f32 {
86 return static_cast<f32>(max(rgb[0], rgb[1], rgb[2]) - min(rgb[0], rgb[1], rgb[2])) / 255.0F;
87 }
88
89 [[nodiscard]] constexpr auto h(std::array<u8, 3> const& rgb) -> f32 {
90 return chroma(rgb) == 0.0F
91 ? 0.0F
92 : (
93 max(rgb[0], rgb[1], rgb[2]) == rgb[0]
94 ? static_cast<f32>(rgb[1] - rgb[2]) / 255.0F / chroma(rgb) + (rgb[1] < rgb[2]
95 ? 6.0F
96 : 0.0F)
97 : (max(rgb[0], rgb[1], rgb[2]) == rgb[1]
98 ? static_cast<f32>(rgb[2] - rgb[0]) / 255.0F / chroma(rgb) + 2.0F
99 : (static_cast<f32>(rgb[0] - rgb[1]) / 255.0F / chroma(rgb) + 4.0F))
100 );
101 }
102
103 [[nodiscard]] constexpr auto lightness(std::array<u8, 3> const& rgb) -> f32 {
104 return static_cast<f32>(min(rgb[0], rgb[1], rgb[2]) + max(rgb[0], rgb[1], rgb[2])) / 510.0F;
105 }

109 ? std::round(h * 60.0F) + 360.0F

314 [[nodiscard]] static constexpr auto from_rgb(u8 const r, u8 const g, u8 const b) -> hsv_t {
315 auto self = hsv_t(0, 0, 0);
316 auto const fr = static_cast<f32>(r) / 255.0F;
317 auto const fg = static_cast<f32>(g) / 255.0F;
318 auto const fb = static_cast<f32>(b) / 255.0F;
319 auto const c_max = detail::max(detail::max(fr, fg), fb);
320 auto const c_min = detail::min(detail::min(fr, fg), fb);
321 auto const delta = c_max - c_min;
322 if(delta > 0.F) {
323 if(c_max == fr)
324 self.h = 60.0F * (std::fmod(((fg - fb) / delta), 6.0F));
325 else if(c_max == fg)
326 self.h = 60.0F * (((fb - fr) / delta) + 2.0F);
327 else if(c_max == fb)
328 self.h = 60.0F * (((fr - fg) / delta) + 4.0F);
329 if(c_max > 0.F)
330 self.s = delta / c_max;
331 else
332 self.s = 0.F;
333 self.v = c_max;
334 } else {
335 self.h = 0.F;
336 self.s = 0.F;
337 self.v = c_max;
338 }
339 if(self.h < 0)
340 self.h += 360;
341 return self;
342 }
