81 [[nodiscard]] constexpr auto a(f32 const saturation, f32 const lightness) -> f32 {
82 return saturation * min(lightness, 1.0F - lightness);

85 [[nodiscard]] constexpr auto chroma(std::array<u8, 3> const& rgb) -> f32 {
86 return static_cast<f32>(max(rgb[0], rgb[1], rgb[2]) - min(rgb[0], rgb[1], rgb[2])) / 255.0F;
87 }
88
89 [[nodiscard]] constexpr auto h(std::array<u8, 3> const& rgb) -> f32 {
90 return chroma(rgb) == 0.0F
91 ? 0.0F
92 : (
93 max(rgb[0], rgb[1], rgb[2]) == rgb[0]
94 ? static_cast<f32>(rgb[1] - rgb[2]) / 255.0F / chroma(rgb) + (rgb[1] < rgb[2]
95 ? 6.0F
96 : 0.0F)
97 : (max(rgb[0], rgb[1], rgb[2]) == rgb[1]
98 ? static_cast<f32>(rgb[2] - rgb[0]) / 255.0F / chroma(rgb) + 2.0F
99 : (static_cast<f32>(rgb[0] - rgb[1]) / 255.0F / chroma(rgb) + 4.0F))
100 );
101 }
102
103 [[nodiscard]] constexpr auto lightness(std::array<u8, 3> const& rgb) -> f32 {
104 return static_cast<f32>(min(rgb[0], rgb[1], rgb[2]) + max(rgb[0], rgb[1], rgb[2])) / 510.0F;
105 }

109 ? std::round(h * 60.0F) + 360.0F
110 : std::round(h * 60.0F));

243 struct hsv_t : public traits::formattable<color::hsv_t, char>
244 {
245 u16 h = 0.0F;
246 f32 s = 0.0F;
247 f32 v = 0.0F;
248
253 constexpr hsv_t(const u16 h, const f32 s, const f32 v)
254 : h(h)
255 , s(s)
256 , v(v)
257 {}
258
260 [[nodiscard]] constexpr auto to_rgb() const -> std::array<u8, 3> {
261 auto rgb = std::array<u8, 3>();
262 auto tmp_rgb = std::array<f32, 3>();
263 const auto c = this->v * this->s ;
264 const auto x = c * (1.0F - std::abs(std::fmod(static_cast<f32>(this->h) / 60.0F, 2.0F) - 1.0F));
265 const auto m = this->v - c;
266 if(static_cast<f32>(this->h)< 60.F)
267 tmp_rgb = {c, x ,0};
268 else if(static_cast<f32>(this->h) < 120.F)
269 tmp_rgb = {x, c, 0};
270 else if(static_cast<f32>(this->h) < 180.F)
271 tmp_rgb = {0, c, x};
272 else if(static_cast<f32>(this->h) < 240.F)
273 tmp_rgb = {0, x, c};
274 else if(static_cast<f32>(this->h) < 300.F)
275 tmp_rgb = {x, 0, c};
276 else if(static_cast<f32>(this->h)< 360.F)
277 tmp_rgb = {c, 0, x};
278
279 rgb[0] = static_cast<u8>((tmp_rgb[0] + m) * 255.0F);
280 rgb[1] = static_cast<u8>((tmp_rgb[1] + m) * 255.0F);
281 rgb[2] = static_cast<u8>((tmp_rgb[2] + m) * 255.0F);
282 return rgb;
283 }
284
289 [[nodiscard]] static constexpr auto from_rgb(u8 const r, u8 const g, u8 const b) -> hsv_t {
290 auto self = hsv_t(0, 0, 0);
291 auto const fr = static_cast<f32>(r) / 255.0F;
292 auto const fg = static_cast<f32>(g) / 255.0F;
293 auto const fb = static_cast<f32>(b) / 255.0F;
294 auto const c_max = detail::max(detail::max(fr, fg), fb);
295 auto const c_min = detail::min(detail::min(fr, fg), fb);
296 auto const delta = c_max - c_min;
297
298 if (delta <= 0.F)
299 self.h = 0.F;
300 else {
301 if(c_max == fr)
302 self.h = std::lround(60.0F * std::fmod((fg - fb) / delta, 6.0F));
303 else if(c_max == fg)
304 self.h = std::lround(60.0F * ((fb - fr) / delta + 2.F));
305 else if(c_max == fb)
306 self.h = std::lround(60.0 * ((fr - fg) / delta + 4.F));
307 }
308
309 self.s = c_max == 0.F ? 0.F : delta / c_max;
310 self.v = c_max;
311
312 return self;
313 }

319 return fmt::format("({} {} {})", this->h, this->s, this->v);
320 }
