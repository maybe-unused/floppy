78 [[nodiscard]] constexpr auto a(f32 const saturation, f32 const lightness) -> f32 {
79 return saturation * min(lightness, 1.0F - lightness);

82 [[nodiscard]] constexpr auto chroma(std::array<u8, 3> const& rgb) -> f32 {
83 return static_cast<f32>(max(rgb[0], rgb[1], rgb[2]) - min(rgb[0], rgb[1], rgb[2])) / 255.0F;
84 }
85
86 [[nodiscard]] constexpr auto h(std::array<u8, 3> const& rgb) -> f32 {
87 return chroma(rgb) == 0.0F
88 ? 0.0F
89 : (
90 max(rgb[0], rgb[1], rgb[2]) == rgb[0]
91 ? static_cast<f32>(rgb[1] - rgb[2]) / 255.0F / chroma(rgb) + (rgb[1] < rgb[2]
92 ? 6.0F
93 : 0.0F)
94 : (max(rgb[0], rgb[1], rgb[2]) == rgb[1]
95 ? static_cast<f32>(rgb[2] - rgb[0]) / 255.0F / chroma(rgb) + 2.0F
96 : (static_cast<f32>(rgb[0] - rgb[1]) / 255.0F / chroma(rgb) + 4.0F))
97 );
98 }
99
100 [[nodiscard]] constexpr auto lightness(std::array<u8, 3> const& rgb) -> f32 {
101 return static_cast<f32>(min(rgb[0], rgb[1], rgb[2]) + max(rgb[0], rgb[1], rgb[2])) / 510.0F;
102 }

308 [[nodiscard]] static constexpr auto from_rgb(u8 const r, u8 const g, u8 const b) -> hsv_t {
309 auto self = hsv_t(0, 0, 0);
310 auto const fr = static_cast<f32>(r) / 255.0F;
311 auto const fg = static_cast<f32>(g) / 255.0F;
312 auto const fb = static_cast<f32>(b) / 255.0F;
313 auto const c_max = detail::max(detail::max(fr, fg), fb);
314 auto const c_min = detail::min(detail::min(fr, fg), fb);
315 auto const delta = c_max - c_min;
316 if(delta > 0.F) {
317 if(c_max == fr)
318 self.h = 60.0F * (std::fmod(((fg - fb) / delta), 6.0F));
319 else if(c_max == fg)
320 self.h = 60.0F * (((fb - fr) / delta) + 2.0F);
321 else if(c_max == fb)
322 self.h = 60.0F * (((fr - fg) / delta) + 4.0F);
323 if(c_max > 0.F)
324 self.s = delta / c_max;
325 else
326 self.s = 0.F;
327 self.v = c_max;
328 } else {
329 self.h = 0.F;
330 self.s = 0.F;
331 self.v = c_max;
332 }
333 if(self.h < 0)
334 self.h += 360;
335 return self;
336 }
