83 [[nodiscard]] constexpr auto a(f32 const saturation, f32 const lightness) -> f32 {
84 return saturation * min(lightness, 1.0F - lightness);

87 [[nodiscard]] constexpr auto chroma(std::array<u8, 3> const& rgb) -> f32 {
88 return static_cast<f32>(max(rgb[0], rgb[1], rgb[2]) - min(rgb[0], rgb[1], rgb[2])) / 255.0F;
89 }
90
91 [[nodiscard]] constexpr auto h(std::array<u8, 3> const& rgb) -> f32 {
92 return chroma(rgb) == 0.0F
93 ? 0.0F
94 : (
95 max(rgb[0], rgb[1], rgb[2]) == rgb[0]
96 ? static_cast<f32>(rgb[1] - rgb[2]) / 255.0F / chroma(rgb) + (rgb[1] < rgb[2]
97 ? 6.0F
98 : 0.0F)
99 : (max(rgb[0], rgb[1], rgb[2]) == rgb[1]
100 ? static_cast<f32>(rgb[2] - rgb[0]) / 255.0F / chroma(rgb) + 2.0F
101 : (static_cast<f32>(rgb[0] - rgb[1]) / 255.0F / chroma(rgb) + 4.0F))
102 );
103 }
104
105 [[nodiscard]] constexpr auto lightness(std::array<u8, 3> const& rgb) -> f32 {
106 return static_cast<f32>(min(rgb[0], rgb[1], rgb[2]) + max(rgb[0], rgb[1], rgb[2])) / 510.0F;
107 }

109 [[nodiscard]] constexpr auto hue(f32 const h) -> u16 {
110 return static_cast<u16>(std::round(h * 60.0F) < 0.0F
111 ? std::round(h * 60.0F) + 360.0F
112 : std::round(h * 60.0F));

242 struct hsv_t : public traits::formattable<color::hsv_t, char>
243 {
244 u16 h = 0.0F;
245 f32 s = 0.0F;
246 f32 v = 0.0F;
247
252 constexpr hsv_t(const u16 h, const f32 s, const f32 v)
253 : h(h)
254 , s(s)
255 , v(v)
256 {}
257
259 [[nodiscard]] constexpr auto to_rgb() const -> std::array<u8, 3> {
260 auto rgb = std::array<u8, 3>();
261 auto tmp_rgb = std::array<f32, 3>();
262 const auto c = this->v * this->s ;
263 const auto x = c * (1.0F - std::abs(std::fmod(static_cast<f32>(this->h) / 60.0F, 2.0F) - 1.0F));
264 const auto m = this->v - c;
265 if(static_cast<f32>(this->h)< 60.F)
266 tmp_rgb = {c, x ,0};
267 else if(static_cast<f32>(this->h) < 120.F)
268 tmp_rgb = {x, c, 0};
269 else if(static_cast<f32>(this->h) < 180.F)
270 tmp_rgb = {0, c, x};
271 else if(static_cast<f32>(this->h) < 240.F)
272 tmp_rgb = {0, x, c};
273 else if(static_cast<f32>(this->h) < 300.F)
274 tmp_rgb = {x, 0, c};
275 else if(static_cast<f32>(this->h)< 360.F)
276 tmp_rgb = {c, 0, x};
277
278 rgb[0] = static_cast<u8>((tmp_rgb[0] + m) * 255.0F);
279 rgb[1] = static_cast<u8>((tmp_rgb[1] + m) * 255.0F);
280 rgb[2] = static_cast<u8>((tmp_rgb[2] + m) * 255.0F);
281 return rgb;
282 }
283
288 [[nodiscard]] static constexpr auto from_rgb(u8 const r, u8 const g, u8 const b) -> hsv_t {
289 auto self = hsv_t(0, 0, 0);
290 auto const fr = static_cast<f32>(r) / 255.0F;
291 auto const fg = static_cast<f32>(g) / 255.0F;
292 auto const fb = static_cast<f32>(b) / 255.0F;
293 auto const c_max = detail::max(detail::max(fr, fg), fb);
294 auto const c_min = detail::min(detail::min(fr, fg), fb);
295 auto const delta = c_max - c_min;
296
297 if (delta <= 0.F)
298 self.h = 0.F;
299 else {
300 if(c_max == fr)
301 self.h = std::lround(60.0F * std::fmod((fg - fb) / delta, 6.0F));
302 else if(c_max == fg)
303 self.h = std::lround(60.0F * ((fb - fr) / delta + 2.F));
304 else if(c_max == fb)
305 self.h = std::lround(60.0 * ((fr - fg) / delta + 4.F));
306 }
307
308 self.s = c_max == 0.F ? 0.F : delta / c_max;
309 self.v = c_max;
310
311 return self;
312 }

317 [[nodiscard]] virtual auto to_string() const -> std::string override {
318 return fmt::format("({} {} {})", this->h, this->s, this->v);
319 }
