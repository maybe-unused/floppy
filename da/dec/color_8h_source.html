82 [[nodiscard]] constexpr auto chroma(std::array<u8, 3> const& rgb) -> f32 {
83 return static_cast<f32>(max(rgb[0], rgb[1], rgb[2]) - min(rgb[0], rgb[1], rgb[2])) / 255.0F;
84 }
85
86 [[nodiscard]] constexpr auto h(std::array<u8, 3> const& rgb) -> f32 {
87 return chroma(rgb) == 0.0F
88 ? 0.0F
89 : (
90 max(rgb[0], rgb[1], rgb[2]) == rgb[0]
91 ? static_cast<f32>(rgb[1] - rgb[2]) / 255.0F / chroma(rgb) + (rgb[1] < rgb[2]
92 ? 6.0F
93 : 0.0F)
94 : (max(rgb[0], rgb[1], rgb[2]) == rgb[1]
95 ? static_cast<f32>(rgb[2] - rgb[0]) / 255.0F / chroma(rgb) + 2.0F
96 : (static_cast<f32>(rgb[0] - rgb[1]) / 255.0F / chroma(rgb) + 4.0F))
97 );
98 }

100 [[nodiscard]] constexpr auto lightness(std::array<u8, 3> const& rgb) -> f32 {
101 return static_cast<f32>(min(rgb[0], rgb[1], rgb[2]) + max(rgb[0], rgb[1], rgb[2])) / 510.0F;
102 }

140 [[nodiscard]] constexpr auto to_rgb() const -> std::array<u8, 3> {
141 return std::array<u8, 3> {
142 static_cast<u8>((this->l - detail::hsl::a(this->s, this->l)
143 * detail::max(-1.0F, detail::min(detail::hsl::k(0, this->h) - 3.0F,
144 9.0F - detail::hsl::k(0, this->h), 1.0F))) * 255.0F),
145 static_cast<u8>((this->l - detail::hsl::a(this->s, this->l)
146 * detail::max(-1.0F, detail::min(detail::hsl::k(8, this->h) - 3.0F,
147 9.0F - detail::hsl::k(8, this->h), 1.0F))) * 255.0F),
148 static_cast<u8>((this->l - detail::hsl::a(this->s, this->l)
149 * detail::max(-1.0F, detail::min(detail::hsl::k(4, this->h) - 3.0F,
