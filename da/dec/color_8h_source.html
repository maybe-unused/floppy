214 [[nodiscard]] constexpr auto hsl() const -> color::hsl {
215 auto constexpr k = 3.0F;
216 auto max = std::max(std::max(this->redf(), this->greenf()), this->bluef());
217 auto min = std::min(std::min(this->redf(), this->greenf()), this->bluef());
218 auto l = (max + min) / 2.0F;
219 auto const delta = max - min;
220 if(math::approx_eq(max, min, k))
221 return {
222 .h = 0.0F,
223 .s = 0.0F,
224 .l = l
225 };
226 auto s = l < 0.5F ? delta / (max + min) : delta / (2.0F - max - min);
227 auto const r2 = (((max - this->redf()) / 6.0F) + (delta / 2.0F)) / delta;
228 auto const g2 = (((max - this->greenf()) / 6.0F) + (delta / 2.0F)) / delta;
229 auto const b2 = (((max - this->bluef()) / 6.0F) + (delta / 2.0F)) / delta;
230 auto h = 0.0F;
231 if(math::approx_eq(max, this->redf(), k))
232 h = b2 - g2;
233 else if(math::approx_eq(max, this->greenf(), k))
234 h = (1.0F / k) + r2 - b2;
235 else
236 h = (2.0F / k) + g2 - r2;
237 if(h < 0.0F)
238 h += 1.0F;
239 if(h > 1.0F)
240 h -= 1.0F;
241 auto const h_deg = std::round(h * 360.0F * 100.0F) / 100.0F;
242 return {
243 .h = h_deg,
244 .s = s,
245 .l = l
246 };
247 }
